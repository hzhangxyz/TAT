154  auto markov(
155  std::uint64_t total_step,
156  std::map<std::string, std::map<std::vector<std::tuple<int, int>>, std::shared_ptr<const Tensor<T>>>> observers,
157  bool calculate_energy = false,
158  bool calculate_gradient = false) {
159  if (calculate_gradient) {
160  calculate_energy = true;
161  }
162  if (calculate_energy) {
163  observers["Energy"] = hamiltonians;
164  }
165  // TODO 如何计算gradient的误差?
166  T sum_of_Es = 0;
167  std::vector<std::vector<Tensor<T>>> holes;
168  std::vector<std::vector<Tensor<T>>> holes_with_Es;
169  std::vector<std::vector<Tensor<T>>> gradient;
170  if (calculate_gradient) {
171  for (auto i = 0; i < M; i++) {
172  auto& row_h = holes.emplace_back();
173  auto& row_e = holes_with_Es.emplace_back();
174  auto& row_g = gradient.emplace_back();
175  for (auto j = 0; j < N; j++) {
176  row_h.push_back(lattice[i][j].same_shape().zero());
177  row_e.push_back(lattice[i][j].same_shape().zero());
178  row_g.push_back(lattice[i][j].same_shape().zero());
179  }
180  }
181  }
182  // 观测量一定是实数
183  auto result = std::map<std::string, std::map<std::vector<std::tuple<int, int>>, real<T>>>();
184  auto result_variance_square = std::map<std::string, std::map<std::vector<std::tuple<int, int>>, real<T>>>();
185  auto result_square = std::map<std::string, std::map<std::vector<std::tuple<int, int>>, real<T>>>();
186  T ws = spin();
187  std::cout << clear_line << "Markov sampling start, total_step=" << total_step << ", dimension=" << dimension_virtual
188  << ", dimension_cut=" << dimension_cut << ", First ws is " << ws << "\n"
189  << std::flush;
190  auto positions_sequence = _markov_sampling_positions_sequence();
191  random::split_seed();
192  for (std::uint64_t step = 0; step < total_step; step++) {
193  ws = _markov_spin(ws, positions_sequence);
194  T Es = 0;
195  for (const auto& [kind, group] : observers) {
196  bool is_energy = kind == "Energy";
197  for (const auto& [positions, tensor] : group) {
198  int body = positions.size();
199  auto current_spin = std::vector<int>();
200  for (auto i = 0; i < body; i++) {
201  current_spin.push_back(spin.configuration[std::get<0>(positions[i])][std::get<1>(positions[i])]);
202  }
203  real<T> value = 0;
204  for (const auto& [spins_out, element] : _find_element(*tensor).at(current_spin)) {
205  auto map = std::map<std::tuple<int, int>, int>();
206  for (auto i = 0; i < body; i++) {
207  map[positions[i]] = spins_out[i];
208  }
209  T wss = spin(map, ws);
210  auto this_term = element * wss / ws;
211  if (is_energy) {
212  // 用于求梯度，这个可能是复数
213  Es += this_term;
214  }
215  value += scalar_to<real<T>>(this_term);
216  }
217  result[kind][positions] += value;
218  result_square[kind][positions] += value * value;
219  }
220  }
221  if (calculate_gradient) {
222  sum_of_Es += Es;
223  for (auto i = 0; i < M; i++) {
224  for (auto j = 0; j < N; j++) {
225  Tensor<T> raw_hole = spin({{i, j}}).edge_rename({{"L0", "L"}, {"R0", "R"}, {"U0", "U"}, {"D0", "D"}});
226  Tensor<T> hole = (raw_hole.conjugate() / conj(ws)).expand({{"P", {spin.configuration[i][j], dimension_physics}}});
227  holes_with_Es[i][j] += hole * Es;
228  holes[i][j] += hole;
229  }
230  }
231  }
232  std::cout << clear_line << "Markov sampling, total_step=" << total_step << ", dimension=" << dimension_virtual
233  << ", dimension_cut=" << dimension_cut << ", step=" << (step + 1) << "\r" << std::flush;
234  }
235  random::merge_seed();
236  // TODO reduce here
237  for (auto& [kind, group] : result) {
238  const auto& group_square = result_square.at(kind);
239  for (auto& [positions, value] : group) {
240  value /= total_step;
241  auto value_square = group_square.at(positions);
242  value_square /= total_step;
243  result_variance_square[kind][positions] = (value_square - value * value) / (total_step - 1);
244  }
245  }
246  if (calculate_energy) {
247  real<T> energy = 0;
248  real<T> energy_variance_square = 0;
249  const auto& energy_variance_square_pool = result_variance_square.at("Energy");
250  for (const auto& [positions, value] : result.at("Energy")) {
251  energy += value;
252  energy_variance_square += energy_variance_square_pool.at(positions);
253  };
254  std::cout << clear_line << "Markov sample done, total_step=" << total_step << ", dimension=" << dimension_virtual
255  << ", dimension_cut=" << dimension_cut << ", Energy=" << energy / (M * N)
256  << " with sigma=" << std::sqrt(energy_variance_square) / (M * N) << "\n"
257  << std::flush;
258  } else {
259  std::cout << clear_line << "Markov sample done, total_step=" << total_step << ", dimension=" << dimension_virtual
260  << ", dimension_cut=" << dimension_cut << "\n"
261  << std::flush;
262  }
263  if (calculate_gradient) {
264  for (auto i = 0; i < M; i++) {
265  for (auto j = 0; j < N; j++) {
266  // TODO reduce gradient
267  gradient[i][j] = 2 * (holes_with_Es[i][j] / total_step) - 2 * (sum_of_Es / total_step) * (holes[i][j] / total_step);
268  }
269  }
270  }
271  return std::make_tuple(std::move(result), std::move(result_variance_square), std::move(gradient));
272  }

275  std::map<std::string, std::map<std::vector<std::tuple<int, int>>, std::shared_ptr<const Tensor<T>>>> observers,
276  bool calculate_energy = false) {
277  std::cout << clear_line << "Ergodic sampling start, dimension=" << dimension_virtual << ", dimension_cut=" << dimension_cut << "\n"
278  << std::flush;
279  if (calculate_energy) {
280  observers["Energy"] = hamiltonians;
281  }
282  auto result = std::map<std::string, std::map<std::vector<std::tuple<int, int>>, real<T>>>();
283  real<T> sum_of_ws_square = 0;
284  std::uint64_t total_step = std::pow(dimension_physics, M * N);
285  for (std::uint64_t step = 0; step < total_step; step++) {
286  _ergodic_spin(step);
287  T ws = spin();
288  sum_of_ws_square += std::norm(ws);
289  for (const auto& [kind, group] : observers) {
290  for (const auto& [positions, tensor] : group) {
291  int body = positions.size();
292  auto current_spin = std::vector<int>();
293  for (auto i = 0; i < body; i++) {
294  current_spin.push_back(spin.configuration[std::get<0>(positions[i])][std::get<1>(positions[i])]);
295  }
296  real<T> value = 0;
297 
298  for (const auto& [spins_out, element] : _find_element(*tensor).at(current_spin)) {
299  auto map = std::map<std::tuple<int, int>, int>();
300  for (auto i = 0; i < body; i++) {
301  map[positions[i]] = spins_out[i];
302  }
303  T wss = spin(map, ws);
304  value += scalar_to<real<T>>(element * wss / ws);
305  }
306  result[kind][positions] += value * std::norm(ws);
307  }
308  }
309  if (calculate_energy) {
310  real<T> energy = 0;
311  for (const auto& [positions, value] : result.at("Energy")) {
312  energy += value;
313  };
314  std::cout << clear_line << "Ergodic sampling, total_step=" << total_step << ", dimension=" << dimension_virtual
315  << ", dimension_cut=" << dimension_cut << ", step=" << (step + 1) << ", Energy=" << energy / (sum_of_ws_square * M * N)
316  << "\r" << std::flush;
317  } else {
318  std::cout << clear_line << "Ergodic sampling, total_step=" << total_step << ", dimension=" << dimension_virtual
319  << ", dimension_cut=" << dimension_cut << ", step=" << (step + 1) << "\r" << std::flush;
320  }
321  }
322  for (auto& [kind, group] : result) {
323  for (auto& [positions, value] : group) {
324  value /= sum_of_ws_square;
325  }
326  }
327  if (calculate_energy) {
328  real<T> energy = 0;
329  for (const auto& [positions, value] : result.at("Energy")) {
330  energy += value;
331  };
332  std::cout << clear_line << "Ergodic sample done, total_step=" << total_step << ", dimension=" << dimension_virtual
333  << ", dimension_cut=" << dimension_cut << ", Energy=" << energy / (M * N) << "\n"
334  << std::flush;
335  } else {
336  std::cout << clear_line << "Ergodic sample done, total_step=" << total_step << ", dimension=" << dimension_virtual
337  << ", dimension_cut=" << dimension_cut << "\n"
338  << std::flush;
339  }
340  return result;
341  }

382  auto _markov_sampling_positions_sequence() const {
383  auto result = std::vector<std::tuple<std::vector<std::tuple<int, int>>, char>>();
384  // 应该不存在常数项的hamiltonians
385  // 双点，横+竖, 单点和横向放在一起
386  for (auto i = 0; i < M; i++) {
387  for (auto j = 0; j < N; j++) {
388  if (auto found = hamiltonians.find({{i, j}}); found != hamiltonians.end()) {
389  result.emplace_back(std::vector<std::tuple<int, int>>{{i, j}}, 'h');
390  }
391  if (auto found = hamiltonians.find({{i, j}, {i, j + 1}}); found != hamiltonians.end()) {
392  result.emplace_back(std::vector<std::tuple<int, int>>{{i, j}, {i, j + 1}}, 'h');
393  }
394  if (auto found = hamiltonians.find({{i, j + 1}, {i, j}}); found != hamiltonians.end()) {
395  result.emplace_back(std::vector<std::tuple<int, int>>{{i, j + 1}, {i, j}}, 'h');
396  }
397  }
398  }
399  for (auto j = 0; j < N; j++) {
400  for (auto i = 0; i < M; i++) {
401  if (auto found = hamiltonians.find({{i, j}, {i + 1, j}}); found != hamiltonians.end()) {
402  result.emplace_back(std::vector<std::tuple<int, int>>{{i, j}, {i + 1, j}}, 'v');

416  T _markov_single_term(
417  T ws,
418  const std::vector<std::tuple<int, int>>& positions,
419  const std::shared_ptr<const Tensor<T>>& hamiltonian,
420  char hint = ' ') {
421 #ifdef LAZY_DEBUG
422  std::clog << "Hopping at ";
423  for (const auto& [x, y] : positions) {
424  std::clog << "(" << x << ", " << y << ") ";
425  }
426  std::clog << "\n";
427 #endif
428  int body = positions.size();
429  auto current_spin = std::vector<int>();
430  for (auto i = 0; i < body; i++) {
431  current_spin.push_back(spin.configuration[std::get<0>(positions[i])][std::get<1>(positions[i])]);
432  }
433  const auto& hamiltonian_elements = _find_hopping_element(*hamiltonian);
434  const auto& possible_hopping = hamiltonian_elements.at(current_spin);
435  if (!possible_hopping.empty()) {
436  int hopping_number = possible_hopping.size();
437  auto random_index = random::uniform<int>(0, hopping_number - 1)();
438  auto iter = possible_hopping.begin();
439  std::advance(iter, random_index);
440  const auto& [spins_new, element] = *iter;
441  auto replacement = std::map<std::tuple<int, int>, int>();
442  for (auto i = 0; i < body; i++) {
443  replacement[positions[i]] = spins_new[i];
444  }
445  T wss = spin(replacement, ws, hint);
446  int hopping_number_s = hamiltonian_elements.at(spins_new).size();
447  T wss_over_ws = wss / ws;
448  real<T> p = std::norm(wss_over_ws) * real<T>(hopping_number) / real<T>(hopping_number_s);
449  if (random::uniform<real<T>>(0, 1)() < p) {
450  ws = wss;
451  for (auto i = 0; i < body; i++) {
452  spin.set(positions[i], spins_new[i]);
453  }
454  }
455  }
456  return ws;
457  }

464  return found->second;

468  auto names = std::vector<Name>();
469  auto index = std::vector<int>();
470  for (auto i = 0; i < body; i++) {
471  names.push_back("I" + std::to_string(i));

473  }
474  for (auto i = 0; i < body; i++) {
475  names.push_back("O" + std::to_string(i));
476  index.push_back(0);
477  }
478  while (true) {
479  auto map = std::map<Name, Size>();
480  for (auto i = 0; i < 2 * body; i++) {
481  map[names[i]] = index[i];

526  }
527  while (true) {
528  auto map = std::map<Name, Size>();
529  for (auto i = 0; i < 2 * body; i++) {
530  map[names[i]] = index[i];
531  }

533  if (value != T(0)) {
534  auto spins_in = std::vector<int>();
535  auto spins_out = std::vector<int>();
536  for (auto i = 0; i < body; i++) {
537  spins_in.push_back(index[i]);

540  spins_out.push_back(index[i]);
