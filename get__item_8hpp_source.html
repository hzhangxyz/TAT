68  const auto rank = Rank(core.edges.size());
69  using VectorSymmetry = typename decltype(core.blocks)::key_type;
70  auto symmetries = VectorSymmetry();
71  auto scalar_position = pmr::vector<Size>();
72  auto dimensions = pmr::vector<Size>();
73  symmetries.reserve(rank);
74  scalar_position.reserve(rank);
75  dimensions.reserve(rank);
76  for (auto i = 0; i < rank; i++) {
77  const auto& name = names[i];
78  auto found = position.find(name);
79  if (found == position.end()) {
80  TAT_error("Name not found in position map when finding block and offset");
81  }
82  const auto& [symmetry, index] = found->second;
83  symmetries.push_back(symmetry);

85  dimensions.push_back(core.edges[i].map.at(symmetry));

89  offset *= dimensions[j];
