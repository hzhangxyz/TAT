62  real<T> update(int total_step, real<T> approximate_energy = -0.5) {
63  TAT::mpi.out_one() << clear_line << "Exact update done, total_step=" << total_step << "\n";
64  real<T> total_approximate_energy = std::abs(approximate_energy) * M * N;
65  real<T> energy = 0;
66  for (auto step = 0; step < total_step; step++) {
67  auto temporary_vector = vector.same_shape().zero();
68  for (const auto& [positions, value] : hamiltonians) {
69  auto position_number = positions.size();
70  auto map_in = std::map<Name, Name>();
71  auto map_out = std::map<Name, Name>();
72  for (auto i = 0; i < position_number; i++) {
73  Name physics_name = "P-" + std::to_string(std::get<0>(positions[i])) + "-" + std::to_string(std::get<1>(positions[i]));
74  map_in["I" + std::to_string(i)] = physics_name;
75  map_out["O" + std::to_string(i)] = physics_name;
76  }
77  temporary_vector += vector.contract_all_edge(value->edge_rename(map_in)).edge_rename(map_out);
78  }
79  vector *= total_approximate_energy;
80  vector -= temporary_vector;
81  // v <- a v - H v = (a - H) v => E = a - v'/v
82  real<T> norm_max = vector.template norm<-1>();
83  energy = total_approximate_energy - norm_max;
84  vector /= norm_max;
85  TAT::mpi.out_one() << clear_line << "Exact updating, total_step=" << total_step << ", step=" << (step + 1)
86  << ", Energy=" << energy / (M * N) << "\r";
87  }
88  TAT::mpi.out_one() << clear_line << "Exact update done, total_step=" << total_step << ", Energy=" << energy / (M * N) << "\n";
89  return energy / (M * N);
90  }

97  real<T> observe(const std::vector<std::tuple<int, int>>& positions, const Tensor<C>& observer, bool calculate_denominator) const {
98  using common = std::common_type_t<C, T>;
99  auto common_vector = vector.template to<common>();
100  auto common_observer = observer.template to<common>();
101  auto position_number = positions.size();
102  auto map_in = std::map<Name, Name>();
103  auto map_out = std::map<Name, Name>();
104  for (auto i = 0; i < position_number; i++) {
105  Name physics_name = "P-" + std::to_string(std::get<0>(positions[i])) + "-" + std::to_string(std::get<1>(positions[i]));
106  map_in["I" + std::to_string(i)] = physics_name;
107  map_out["O" + std::to_string(i)] = physics_name;
108  }
109  auto numerator = common_vector.contract_all_edge(common_observer.edge_rename(map_in))
110  .edge_rename(map_out)
111  .contract_all_edge(common_vector)
112  .template to<real<T>>();
113  if (calculate_denominator) {
114  return real<T>(numerator) / denominator();
115  } else {
116  return real<T>(numerator);
117  }
118  }
