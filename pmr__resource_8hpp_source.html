125  void release() {
126  for (auto [buffer, size] : m_buffer_list) {
127  m_upstream->deallocate(buffer, size, alignof(std::max_align_t));
128  }
129  m_buffer_list.clear();

131  m_current_buffer_size = m_initial_buffer_size;
132  m_next_buffer_size = default_next_buffer_size;

135  std::size_t remaining_storage(std::size_t alignment, std::size_t& aligner) {
136  const std::size_t up_alignment_minus1 = alignment - 1u;
137  const std::size_t up_alignment_mask = ~up_alignment_minus1;
138  const std::size_t up_addr = std::size_t(m_current_buffer);
139  const std::size_t up_aligned_addr = (up_addr + up_alignment_minus1) & up_alignment_mask;
140  aligner = std::size_t(up_aligned_addr - up_addr);
141  return m_current_buffer_size <= aligner ? 0u : m_current_buffer_size - aligner;
142  }

144  void* allocate_from_current(std::size_t aligner, std::size_t bytes) {
145  std::byte* result = (std::byte*)m_current_buffer + aligner;
146  m_current_buffer = result + bytes;
147  m_current_buffer_size -= aligner + bytes;
148  return result;
149  }

151  void* do_allocate(std::size_t bytes, std::size_t alignment) override {
152  std::size_t aligner = 0u;
153  if (remaining_storage(alignment, aligner) < bytes) {
154  aligner = 0u;
155  increase_next_size_to(bytes);
156  m_current_buffer = m_upstream->allocate(m_next_buffer_size, alignof(std::max_align_t));
157  m_current_buffer_size = m_next_buffer_size;
158  m_buffer_list.push_front({m_current_buffer, m_current_buffer_size});
159  increase_next_size();
160  }
161  return allocate_from_current(aligner, bytes);
162  }
