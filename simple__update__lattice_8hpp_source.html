54  void update(int total_step, real<T> delta_t, Size new_dimension) {
55  if (new_dimension) {
56  dimension_virtual = new_dimension;
57  }
58  TAT::mpi.out_one() << clear_line << "Simple updating start, total_step=" << total_step << ", dimension=" << dimension_virtual
59  << ", delta_t=" << delta_t << "\n";
60  std::map<const Tensor<T>*, std::shared_ptr<const Tensor<T>>> updater_pool;
61  std::map<std::vector<std::tuple<int, int>>, std::shared_ptr<const Tensor<T>>> updater;
62  for (const auto& [positions, term] : hamiltonians) {
63  auto position_number = positions.size();
64  auto term_pointer = &*term;
65  if (auto found = updater_pool.find(term_pointer); found == updater_pool.end()) {
66  auto pairs = std::set<std::tuple<Name, Name>>();
67  for (auto i = 0; i < position_number; i++) {
68  pairs.insert({"I" + std::to_string(i), "O" + std::to_string(i)});
69  }
70  updater[positions] = updater_pool[term_pointer] = std::make_shared<const Tensor<T>>((-delta_t * *term).exponential(pairs, 8));
71  } else {
72  updater[positions] = found->second;
73  }
74  }
75  auto positions_sequence = _simple_update_positions_sequence();
76  for (auto step = 0; step < total_step; step++) {
77  for (auto iter = positions_sequence.begin(); iter != positions_sequence.end(); ++iter) {
78  _single_group_simple_update(*iter, updater);
79  }
80  for (auto iter = positions_sequence.rbegin(); iter != positions_sequence.rend(); ++iter) {
81  _single_group_simple_update(*iter, updater);
82  }
83  TAT::mpi.out_one() << clear_line << "Simple updating, total_step=" << total_step << ", dimension=" << dimension_virtual
84  << ", delta_t=" << delta_t << ", step=" << (step + 1) << "\r";
85  }
86  TAT::mpi.out_one() << clear_line << "Simple update done, total_step=" << total_step << ", dimension=" << dimension_virtual
87  << ", delta_t=" << delta_t << "\n";
88  }

92  const std::map<std::vector<std::tuple<int, int>>, std::shared_ptr<const Tensor<T>>>& updater) {

104  for (const auto& [x, y] : positions) {

113  static std::vector<std::tuple<char, int, int>> _get_related_environment(const std::vector<std::tuple<int, int>>& positions) {
114  if (positions.size() == 1) {
115  return {};
116  }
117  if (positions.size() == 2) {
118  const auto& [x1, y1] = positions[0];
119  const auto& [x2, y2] = positions[1];
120  if (x1 == x2 && y1 + 1 == y2) {
121  return {{'R', x1, y1}};
122  }
123  if (x1 == x2 && y1 == y2 + 1) {
124  return {{'R', x2, y2}};
125  }
126  if (x1 + 1 == x2 && y1 == y2) {
127  return {{'D', x1, y1}};
128  }
129  if (x1 == x2 + 1 && y1 == y2) {
130  return {{'D', x2, y2}};
131  }
132  }
133  throw NotImplementedError("Unsupported environment style");
134  }

136  auto _simple_update_positions_sequence() const {
137  // 以后simple update如果要并行，下面之中每类内都是无依赖的
138  auto result = std::vector<std::vector<std::vector<std::tuple<int, int>>>>();
139  // 应该不存在常数项的hamiltonians
140  // 单点
141  auto& group0 = result.emplace_back();
142  for (const auto& [positions, tensor] : hamiltonians) {
143  if (positions.size() == 1) {
144  group0.push_back(positions);
145  }
146  }
147  // 双点，分成四大类+其他
148  auto& group1 = result.emplace_back();
149  for (const auto& [positions, tensor] : hamiltonians) {
150  if (positions.size() == 2) {
151  const auto& [x1, y1] = positions[0];
152  const auto& [x2, y2] = positions[1];
153  if (x1 == x2 && ((y1 + 1 == y2 && y1 % 2 == 0) || (y1 == y2 + 1 && y2 % 2 == 0))) {
154  group1.push_back(positions);
155  }
156  }
157  }
158  auto& group2 = result.emplace_back();
159  for (const auto& [positions, tensor] : hamiltonians) {
160  if (positions.size() == 2) {
161  const auto& [x1, y1] = positions[0];
162  const auto& [x2, y2] = positions[1];
163  if (x1 == x2 && ((y1 + 1 == y2 && y1 % 2 == 1) || (y1 == y2 + 1 && y2 % 2 == 1))) {
164  group2.push_back(positions);
165  }
166  }
167  }
168  auto& group3 = result.emplace_back();
169  for (const auto& [positions, tensor] : hamiltonians) {
170  if (positions.size() == 2) {
171  const auto& [x1, y1] = positions[0];
172  const auto& [x2, y2] = positions[1];
173  if (y1 == y2 && ((x1 + 1 == x2 && x1 % 2 == 0) || (x1 == x2 + 1 && x2 % 2 == 0))) {
174  group3.push_back(positions);
175  }
176  }
177  }
178  auto& group4 = result.emplace_back();
179  for (const auto& [positions, tensor] : hamiltonians) {
180  if (positions.size() == 2) {
181  const auto& [x1, y1] = positions[0];
182  const auto& [x2, y2] = positions[1];
183  if (y1 == y2 && ((x1 + 1 == x2 && x1 % 2 == 1) || (x1 == x2 + 1 && x2 % 2 == 1))) {
184  group4.push_back(positions);
185  }
186  }
187  }
188  // 其他类型和更多的格点暂时不支持
189  auto total_size = 0;
190  for (const auto& group : result) {
191  total_size += group.size();
192  }
193  if (total_size != hamiltonians.size()) {
194  throw NotImplementedError("Unsupported simple update style");
195  }
196  return result;

199  void _single_term_simple_update(const std::vector<std::tuple<int, int>>& positions, const Tensor<T>& updater) {
200  if (positions.size() == 0) {
201  } else if (positions.size() == 1) {
202  _single_term_simple_update_single_site(positions[0], updater);
203  return;
204  } else if (positions.size() == 2) {
205  const auto& position_1 = positions[0];
206  const auto& position_2 = positions[1];
207  if (std::get<0>(position_1) == std::get<0>(position_2)) {
208  if (std::get<1>(position_1) == std::get<1>(position_2) + 1) {
209  _single_term_simple_update_horizontal_bond(position_2, updater);
210  return;
211  }
212  if (std::get<1>(position_1) + 1 == std::get<1>(position_2)) {
213  _single_term_simple_update_horizontal_bond(position_1, updater);
214  return;
215  }
216  }
217  if (std::get<1>(position_1) == std::get<1>(position_2)) {
218  if (std::get<0>(position_1) == std::get<0>(position_2) + 1) {
219  _single_term_simple_update_vertical_bond(position_2, updater);
220  return;
221  }
222  if (std::get<0>(position_1) + 1 == std::get<0>(position_2)) {
223  _single_term_simple_update_vertical_bond(position_1, updater);
224  return;
225  }
226  }
227  }
228  throw NotImplementedError("Unsupported simple update style");
229  }

236  void _single_term_simple_update_horizontal_bond(const std::tuple<int, int>& position, const Tensor<T>& updater) {
237  auto [i, j] = position;
238  auto left = lattice[i][j];
239  left = try_multiple(left, i, j, 'L');
240  left = try_multiple(left, i, j, 'U');
241  left = try_multiple(left, i, j, 'D');
242  left = try_multiple(left, i, j, 'R');
243  auto right = lattice[i][j + 1];
244  right = try_multiple(right, i, j + 1, 'U');
245  right = try_multiple(right, i, j + 1, 'D');
246  right = try_multiple(right, i, j + 1, 'R');
247  auto [left_q, left_r] = left.qr('r', {"P", "R"}, "R", "L");
248  auto [right_q, right_r] = right.qr('r', {"P", "L"}, "L", "R");
249  auto [u, s, v] = left_r.edge_rename({{"P", "P0"}})
250  .contract(right_r.edge_rename({{"P", "P1"}}), {{"R", "L"}})
251  .contract(updater, {{"P0", "I0"}, {"P1", "I1"}})
252  .svd({"L", "O0"}, "R", "L", dimension_virtual);
253  s /= s.template norm<-1>();
254  environment[{'R', i, j}] = std::move(s);
255  u = u.contract(left_q, {{"L", "R"}}).edge_rename({{"O0", "P"}});
256  u = try_multiple(u, i, j, 'L', true);
257  u = try_multiple(u, i, j, 'U', true);
258  u = try_multiple(u, i, j, 'D', true);
259  u /= u.template norm<-1>();
260  lattice[i][j] = std::move(u);
261  v = v.contract(right_q, {{"R", "L"}}).edge_rename({{"O1", "P"}});
262  v = try_multiple(v, i, j + 1, 'U', true);
263  v = try_multiple(v, i, j + 1, 'D', true);
264  v = try_multiple(v, i, j + 1, 'R', true);
265  v /= v.template norm<-1>();
266  lattice[i][j + 1] = std::move(v);
267  }

269  void _single_term_simple_update_vertical_bond(const std::tuple<int, int>& position, const Tensor<T>& updater) {
270  auto [i, j] = position;
271  auto up = lattice[i][j];
272  up = try_multiple(up, i, j, 'L');
273  up = try_multiple(up, i, j, 'U');
274  up = try_multiple(up, i, j, 'D');
275  up = try_multiple(up, i, j, 'R');
276  auto down = lattice[i + 1][j];
277  down = try_multiple(down, i + 1, j, 'L');
278  down = try_multiple(down, i + 1, j, 'D');
279  down = try_multiple(down, i + 1, j, 'R');
280  auto [up_q, up_r] = up.qr('r', {"P", "D"}, "D", "U");
281  auto [down_q, down_r] = down.qr('r', {"P", "U"}, "U", "D");
282  auto [u, s, v] = up_r.edge_rename({{"P", "P0"}})
283  .contract(down_r.edge_rename({{"P", "P1"}}), {{"D", "U"}})
284  .contract(updater, {{"P0", "I0"}, {"P1", "I1"}})
285  .svd({"U", "O0"}, "D", "U", dimension_virtual);
286  s /= s.template norm<-1>();
287  environment[{'D', i, j}] = std::move(s);
288  u = u.contract(up_q, {{"U", "D"}}).edge_rename({{"O0", "P"}});
289  u = try_multiple(u, i, j, 'L', true);
290  u = try_multiple(u, i, j, 'U', true);
291  u = try_multiple(u, i, j, 'R', true);
292  u /= u.template norm<-1>();
293  lattice[i][j] = std::move(u);
294  v = v.contract(down_q, {{"D", "U"}}).edge_rename({{"O1", "P"}});
295  v = try_multiple(v, i + 1, j, 'L', true);
296  v = try_multiple(v, i + 1, j, 'D', true);
297  v = try_multiple(v, i + 1, j, 'R', true);
298  v /= v.template norm<-1>();
299  lattice[i + 1][j] = std::move(v);
300  }

302  auto try_multiple(Tensor<T> tensor, int i, int j, char direction, bool division = false) const {
303  if (direction == 'L') {
304  if (auto found = environment.find(std::tuple<char, int, int>('R', i, j - 1)); found != environment.end()) {
305  return tensor.multiple(found->second, "L", 'v', division);
306  }
307  }
308  if (direction == 'U') {
309  if (auto found = environment.find(std::tuple<char, int, int>('D', i - 1, j)); found != environment.end()) {
310  return tensor.multiple(found->second, "U", 'v', division);
311  }
312  }
313  if (direction == 'D') {
314  if (auto found = environment.find(std::tuple<char, int, int>('D', i, j)); found != environment.end()) {
315  return tensor.multiple(found->second, "D", 'u', division);
316  }
317  }
318  if (direction == 'R') {
319  if (auto found = environment.find(std::tuple<char, int, int>('R', i, j)); found != environment.end()) {
320  return tensor.multiple(found->second, "R", 'u', division);
321  }
322  }
323  return tensor;
324  }
