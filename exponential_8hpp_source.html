94  template<typename ScalarType>
95  void matrix_exponential(Size n, ScalarType* A, ScalarType* F, int q) {
96  int int_n = n;
97  // j = max(0, 1+floor(log2(|A|_inf)))
98  auto j = std::max(0, 1 + int(std::log2(max_of_abs(A, n))));
99  // A = A/2^j
100  ScalarType parameter = ScalarType(1) / ScalarType(1 << j);
101  for (Size i = 0; i < n * n; i++) {
102  A[i] *= parameter;
103  }
104  // D=I, N=I, X=I, c=1
105  vector<ScalarType> D(n * n);
106  initialize_identity_matrix(D.data(), n);
107  vector<ScalarType> N(n * n);
108  initialize_identity_matrix(N.data(), n);
109  vector<ScalarType> X1(n * n);
110  initialize_identity_matrix(X1.data(), n);
111  vector<ScalarType> X2(n * n);
112  ScalarType c = 1;
113  // for k=1:q
114  const ScalarType alpha = 1;
115  const ScalarType beta = 0;
116  for (auto k = 1; k <= q; k++) {
117  // c = (c*(q-k+1))/((2*q-k+1)*k)
118  c = (c * ScalarType(q - k + 1)) / ScalarType((2 * q - k + 1) * k);
119  // X = A@X, N=N+c*X, D=D+(-1)^k*c*X
120  auto& X_old = k % 2 == 1 ? X1 : X2;
121  auto& X_new = k % 2 == 0 ? X1 : X2;
122  // new = A @ old
123  // new.T = old.T @ A.T
124  gemm<ScalarType>("N", "N", &int_n, &int_n, &int_n, &alpha, X_old.data(), &int_n, A, &int_n, &beta, X_new.data(), &int_n);
125  ScalarType d = k % 2 == 0 ? c : -c;
126  for (Size i = 0; i < n * n; i++) {
127  auto x = X_new[i];
128  N[i] += c * x;
129  D[i] += d * x;
130  }
131  }
132  // solve D@F=N for F
133  vector<ScalarType> F1(n * n);
134  vector<ScalarType> F2(n * n);
135  auto* R = j == 0 ? F : F1.data();
136  // D@R=N
137  linear_solve<ScalarType>(n, D.data(), n, N.data(), R);
138  // for k=1:j
139  for (auto k = 1; k <= j; k++) {
140  // F = F@F
141  const auto* F_old = k % 2 == 1 ? F1.data() : F2.data();
142  auto* F_new = k == j ? F : k % 2 == 0 ? F1.data() : F2.data();
143  // new = old * old
144  // new.T = old.T * old.T
145  gemm<ScalarType>("N", "N", &int_n, &int_n, &int_n, &alpha, F_old, &int_n, F_old, &int_n, &beta, F_new, &int_n);
146  }
147  }
